% define shortcuts and general text segments

% \nl inserts an empty line
\define[]\nl{\emptylines[1]}

% \archiveNotes inserts generic archive.org notes that shall be available in every entry
\define[]\archiveNotes%
{%
	for more information take a look at/para mais informações, veja\nl%
	\from[rtc]%
}

\define[]\archiveDescrEn%
{%
	[en]\nl%
}

\define[]\archiveDescrPt%
{%
	[pt]\nl%
}
