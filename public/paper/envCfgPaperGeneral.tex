\startenvironment envCfgPaperGeneral

% use utf-8 encoding
\enableregime	[utf-8]

% setup fonts
\usetypescript[adventor][ec]
%\usetypescript[heros][ec]
%\usetypescript[schola][ec]
\setupbodyfont[adventor, 10pt]

    color=green:6,
    style=\ss]
